11/// ロリハ
12/// [`Monoid`](ac_library::Monoid)として使う。
13pub enum RollingHash {}
14
15impl RollingHash {
16    const MOD: u64 = (1 << 61) - 1;
17
18    /// 文字`c`からロリハを生成する。
19    pub fn new(c: char) -> <Self as Monoid>::S {
20        let (b1, b2) = Self::bases();
21        (b1, b2, c as u64, c as u64)
22    }
23
24    fn bases() -> (u64, u64) {
25        thread_local! {
26            static BASES: OnceCell<(u64, u64)> = const {OnceCell::new()};
27        }
28        BASES.with(|bases| {
29            *bases.get_or_init(|| {
30                let mut rng = SmallRng::from_os_rng();
31                let range = Uniform::new(0, RollingHash::MOD).unwrap();
32                (range.sample(&mut rng), range.sample(&mut rng))
33            })
34        })
35    }
36
37    #[inline]
38    fn add(a: u64, b: u64) -> u64 {
39        let c = a + b;
40        if c >= Self::MOD { c - Self::MOD } else { c }
41    }
42
43    #[inline]
44    fn neg(a: u64) -> u64 {
45        if a == 0 { 0 } else { Self::MOD - a }
46    }
47
48    #[inline]
49    fn sub(a: u64, b: u64) -> u64 {
50        Self::add(a, Self::neg(b))
51    }
52
53    #[inline]
54    fn mul(a: u64, b: u64) -> u64 {
55        let c = (a as u128) * (b as u128);
56        let m = Self::MOD as u128;
57        let c = (c >> 61) + (c & m);
58        (if c >= m { c - m } else { c }) as u64
59    }
60}
61
62impl Monoid for RollingHash {
63    /// (base1, base2, hash1, hash2)
64    type S = (u64, u64, u64, u64);
65    fn identity() -> Self::S {
66        (1, 1, 0, 0)
67    }
68    fn binary_operation(a: &Self::S, b: &Self::S) -> Self::S {
69        (
70            Self::mul(a.0, b.0),
71            Self::mul(a.1, b.1),
72            Self::add(Self::mul(a.2, b.0), b.2),
73            Self::add(Self::mul(a.3, b.1), b.3),
74        )
75    }
76}

78/// ロリハによる文字列検索のための構造体。
79/// ```
80/// use qitoy_rolling_hash::RhVec;
81/// let s = "abracadabra";
82/// let h: RhVec = s.chars().collect();
83/// assert_eq!(h.get(0..4), h.get(7..11)); // "abra"
84/// assert_eq!(h.get(0..0), h.get(4..4)); // ""
85/// assert_ne!(h.get(0..4), h.get(4..8)); // "abra", "cada"
86/// ```
87pub struct RhVec {
88    data: Vec<<RollingHash as Monoid>::S>,
89}
90
91impl RhVec {
92    pub fn get(&self, range: std::ops::Range<usize>) -> <RollingHash as Monoid>::S {
93        let (l, r) = (range.start, range.end);
94        let b = self.data[r - l];
95        let (l, r) = (self.data[l], self.data[r]);
96        (
97            b.0,
98            b.1,
99            RollingHash::sub(r.2, RollingHash::mul(l.2, b.0)),
100            RollingHash::sub(r.3, RollingHash::mul(l.3, b.1)),
101        )
102    }
103}
104
105impl FromIterator<char> for RhVec {
106    fn from_iter<T: IntoIterator<Item = char>>(iter: T) -> Self {
107        let e = RollingHash::identity();
108        let data = std::iter::once(e)
109            .chain(iter.into_iter().scan(e, |state, x| {
110                *state = RollingHash::binary_operation(state, &RollingHash::new(x));
111                Some(*state)
112            }))
113            .collect();
114        Self { data }
115    }
