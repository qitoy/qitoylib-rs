3use std::ops::{Add, Mul, Neg, Sub};

5/// モンゴメリ乗算で内部的に使われる型。
6#[derive(Debug, Clone, PartialEq, Eq)]
7pub struct Mvalue {
8    val: u128,
9    m: Montgomery,
10}
11
12impl Mvalue {
13    /// 保有している数値を取り出す。
14    pub fn val(&self) -> u64 {
15        self.m.reduction(self.val) as u64
16    }
17
18    pub fn pow(mut self, mut rhs: u64) -> Self {
19        let mut r = self.m.trans(1);
20        while rhs > 0 {
21            if rhs & 1 == 1 {
22                r = r * self.clone();
23            }
24            self = self.clone() * self;
25            rhs >>= 1;
26        }
27        r
28    }
29}
30
31impl Add for Mvalue {
32    type Output = Self;
33    fn add(self, rhs: Self) -> Self::Output {
34        assert_eq!(self.m, rhs.m);
35        let val = self.val + rhs.val;
36        let m = self.m.m;
37        let val = if val >= m { val - m } else { val };
38        Self { val, m: self.m }
39    }
40}
41
42impl Neg for Mvalue {
43    type Output = Self;
44    fn neg(self) -> Self::Output {
45        if self.val == 0 {
46            self
47        } else {
48            Self {
49                val: self.m.m - self.val,
50                m: self.m,
51            }
52        }
53    }
54}
55
56impl Sub for Mvalue {
57    type Output = Self;
58    fn sub(self, rhs: Self) -> Self::Output {
59        self + (-rhs)
60    }
61}
62
63impl Mul for Mvalue {
64    type Output = Self;
65    fn mul(self, rhs: Self) -> Self::Output {
66        assert_eq!(self.m, rhs.m);
67        let val = self.m.reduction(self.val * rhs.val);
68        Self { val, m: self.m }
69    }
70}

72/// モンゴメリ乗算をするための構造体。
73#[derive(Debug, Clone, PartialEq, Eq)]
74pub struct Montgomery {
75    r: u128,
76    m: u128,
77}
78
79impl Montgomery {
80    /// `modulus`で前計算をする。ただし`modulus`は奇数でなくてはならない。
81    pub fn new(modulus: u64) -> Self {
82        assert!(modulus % 2 != 0);
83        let m = modulus.into();
84        let (mut r, mut t) = (0, 0);
85        for i in 0..64 {
86            if t & 1 == 0 {
87                t += m;
88                r += 1 << i;
89            }
90            t >>= 1;
91        }
92        Self { r, m }
93    }
94
95    /// `val`をモンゴメリ乗算用に変換する。
96    pub fn trans(&self, val: u64) -> Mvalue {
97        let val: u128 = val.into();
98        let val = (val << 64) % self.m;
99        Mvalue {
100            val,
101            m: self.clone(),
102        }
103    }
104
105    fn reduction(&self, val: u128) -> u128 {
106        let mask: u128 = u64::MAX.into();
107        let b = ((val & mask) * self.r) & mask;
108        let c = (val + b * self.m) >> 64;
109        if c >= self.m { c - self.m } else { c }
110    }
111}
