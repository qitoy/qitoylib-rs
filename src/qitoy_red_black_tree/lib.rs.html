6pub trait MAct {
7    /// element type
8    type S: Clone;
9    /// map type
10    type F: Clone + PartialEq;
11    /// identity element
12    fn e() -> Self::S;
13    /// binary operation
14    fn op(a: &Self::S, b: &Self::S) -> Self::S;
15    /// identity map
16    fn id() -> Self::F;
17    /// composition
18    /// `|x| f(g(x))`
19    fn comp(f: &Self::F, g: &Self::F) -> Self::F;
20    /// mapping
21    /// `f(x)`
22    /// * `len` - subtree's len (where subtree's value is x)
23    fn map(f: &Self::F, x: &Self::S, len: usize) -> Self::S;
24}

26#[derive(Clone, Copy, PartialEq, Eq)]

29    Black,

32struct Node<M: MAct> {
33    left: Option<Rc<Node<M>>>,
34    right: Option<Rc<Node<M>>>,
35    color: Color,
36    /// 葉までの黒ノードの数（自身は含まない）
37    rank: usize,
38    /// 部分木の葉ノード数
39    len: usize,
40    val: M::S,
41    lazy: M::F,
42    rev: bool,
43}

87impl<M: MAct> Node<M> {
88    fn new(left: Rc<Self>, right: Rc<Self>, color: Color) -> Rc<Self> {
89        Self {
90            left: Some(left.clone()),
91            right: Some(right.clone()),
92            color,
93            rank: left.rank
94                + match left.color {
95                    Black => 1,
96                    Red => 0,
97                },
98            len: left.len + right.len,
99            val: M::op(&left.val, &right.val),
100            lazy: M::id(),
101            rev: false,
102        }
103        .into()
104    }
105
106    fn leaf(val: M::S) -> Rc<Self> {
107        Self {
108            val,
109            ..Default::default()
110        }
111        .into()
112    }
113
114    #[inline]
115    fn is_leaf(&self) -> bool {
116        self.left.is_none()
117    }
118
119    #[inline]
120    fn left(&self) -> Rc<Self> {
121        self.left.clone().unwrap()
122    }
123
124    #[inline]
125    fn right(&self) -> Rc<Self> {
126        self.right.clone().unwrap()
127    }
128
129    fn push_new(mut self: Rc<Self>) -> Rc<Self> {
130        if !self.is_leaf() && (self.lazy != M::id() || self.rev) {
131            let (left, right) = self.lazy_push().unwrap();
132            let this = Rc::make_mut(&mut self);
133            this.left = Some(left);
134            this.right = Some(right);
135            this.lazy = M::id();
136            this.rev = false;
137        }
138        self
139    }
140
141    fn lazy_push(&self) -> Option<(Rc<Self>, Rc<Self>)> {
142        let Self {
143            left: Some(left),
144            right: Some(right),
145            lazy,
146            rev,
147            ..
148        } = self
149        else {
150            return None;
151        };
152        (
153            left.clone().update(lazy, *rev),
154            right.clone().update(lazy, *rev),
155        )
156            .into()
157    }
158
159    fn update(mut self: Rc<Self>, lazy: &M::F, rev: bool) -> Rc<Self> {
160        if !self.is_leaf() && rev || lazy != &M::id() {
161            let Self {
162                left,
163                right,
164                len,
165                val: v,
166                lazy: l,
167                rev: r,
168                ..
169            } = Rc::make_mut(&mut self);
170            *v = M::map(lazy, v, *len);
171            *l = M::comp(lazy, l);
172            if rev {
173                *r ^= true;
174                std::mem::swap(left, right);
175            }
176        }
177        self
178    }
179
180    #[inline]
181    fn to_black(mut self: Rc<Self>) -> Rc<Self> {
182        Rc::make_mut(&mut self).color = Black;
183        self
184    }
185
186    #[inline]
187    fn merge(self: Rc<Self>, v: Rc<Self>) -> Rc<Self> {
188        let (left, right, _) = Self::merge_sub(self, v);
189        Self::new(left, right, Black)
190    }
191
192    fn merge_sub(a: Rc<Self>, b: Rc<Self>) -> (Rc<Self>, Rc<Self>, Color) {
193        let (a, b) = (a.push_new(), b.push_new());
194        match Ord::cmp(&a.rank, &b.rank) {
195            Less => {
196                let (left, right, color) = Self::merge_sub(a, b.left());
197                if b.color == Black && color == Red && left.color == Red {
198                    if b.right().color == Black {
199                        (left, Self::new(right, b.right(), Red), Black)
200                    } else {
201                        (Self::new(left, right, Black), b.right().to_black(), Red)
202                    }
203                } else {
204                    (Self::new(left, right, color), b.right(), b.color)
205                }
206            }
207            Greater => {
208                let (left, right, color) = Self::merge_sub(a.right(), b);
209                if a.color == Black && color == Red && right.color == Red {
210                    if a.left().color == Black {
211                        (Self::new(a.left(), left, Red), right, Black)
212                    } else {
213                        (a.left().to_black(), Self::new(left, right, Black), Red)
214                    }
215                } else {
216                    (a.left(), Self::new(left, right, color), a.color)
217                }
218            }
219            Equal => (a, b, Red),
220        }
221    }
222
223    fn split(self: Rc<Self>, k: usize) -> (Rc<Self>, Rc<Self>) {
224        let (left, right) = self.lazy_push().unwrap();
225        let len = left.len;
226        match k.cmp(&len) {
227            Less => {
228                let (l, r) = left.split(k);
229                (l, r.merge(right.to_black()))
230            }
231            Greater => {
232                let (l, r) = right.split(k - len);
233                (left.to_black().merge(l), r)
234            }
235            Equal => (left.to_black(), right.to_black()),
236        }
237    }
238
239    /// 0 < l < r < n
240    fn split3(self: Rc<Self>, kl: usize, kr: usize) -> (Rc<Self>, Rc<Self>, Rc<Self>) {
241        let (left, right) = self.lazy_push().unwrap();
242        let len = left.len;
243        match (kl.cmp(&len), kr.cmp(&len)) {
244            (_, Less) => {
245                let (l, m, r) = left.split3(kl, kr);
246                (l, m, r.merge(right.to_black()))
247            }
248            (_, Equal) => {
249                let (l, m) = left.split(kl);
250                (l, m, right.to_black())
251            }
252            (Less, Greater) => {
253                let (l, ml) = left.split(kl);
254                let (mr, r) = right.split(kr - len);
255                (l, ml.merge(mr), r)
256            }
257            (Equal, _) => {
258                let (m, r) = right.split(kr - len);
259                (left.to_black(), m, r)
260            }
261            (Greater, _) => {
262                let (l, m, r) = right.split3(kl - len, kr - len);
263                (left.to_black().merge(l), m, r)
264            }
265        }
266    }
267
268    fn set(self: Rc<Self>, p: usize, val: M::S) -> Rc<Self> {
269        match self.lazy_push() {
270            None => Self::leaf(val),
271            Some((left, right)) => {
272                let len = left.len;
273                let (left, right) = if p < len {
274                    (left.set(p, val), right)
275                } else {
276                    (left, right.set(p - len, val))
277                };
278                Self::new(left, right, self.color)
279            }
280        }
281    }
282
283    fn prod(self: Rc<Self>, l: usize, r: usize, lazy: &M::F, rev: bool) -> M::S {
284        if l == r {
285            return M::e();
286        }
287        if l == 0 && r == self.len {
288            return M::map(lazy, &self.val, self.len);
289        }
290        let Self {
291            left: Some(left),
292            right: Some(right),
293            lazy: la,
294            rev: re,
295            ..
296        } = self.as_ref()
297        else {
298            unreachable!();
299        };
300        let (left, right) = if rev { (right, left) } else { (left, right) };
301        let (lazy, rev) = (&M::comp(lazy, la), rev ^ re);
302        let len = &left.len;
303        match (l.cmp(len), r.cmp(len)) {
304            (_, Less | Equal) => left.clone().prod(l, r, lazy, rev),
305            (Less, Greater) => M::op(
306                &left.clone().prod(l, *len, lazy, rev),
307                &right.clone().prod(0, r - len, lazy, rev),
308            ),
309            (Equal | Greater, _) => right.clone().prod(l - len, r - len, lazy, rev),
310        }
311    }
312
313    fn apply(self: Rc<Self>, l: usize, r: usize, lazy: &M::F) -> Rc<Self> {
314        if l == r {
315            return self;
316        }
317        if l == 0 && r == self.len {
318            return self.update(lazy, false);
319        }
320        let (left, right) = self.lazy_push().unwrap();
321        let len = left.len;
322        let (left, right) = match (l.cmp(&len), r.cmp(&len)) {
323            (_, Less | Equal) => (left.apply(l, r, lazy), right),
324            (Less, Greater) => (left.apply(l, len, lazy), right.apply(0, r - len, lazy)),
325            (Equal | Greater, _) => (left, right.apply(l - len, r - len, lazy)),
326        };
327        Self::new(left, right, self.color)
328    }
329
330    fn dump(self: Rc<Self>, dump: &mut Vec<M::S>) {
331        match self.lazy_push() {
332            None => dump.push(self.val.clone()),
333            Some((left, right)) => {
334                left.dump(dump);
335                right.dump(dump);
336            }
337        }
338    }
339
340    fn debug(self: Rc<Self>) -> String
341    where
342        M::S: Debug,
343    {
344        let dbg = match self.lazy_push() {
345            None => format!("{:?}", self.val),
346            Some((left, right)) => {
347                format!("{} {}", left.debug(), right.debug())
348            }
349        };
350        match self.color {
351            Red => dbg,
352            Black => format!("[{}]", dbg),
353        }
354    }
355}

399impl<M: MAct> RedBlackTree<M> {
400    pub fn new(val: M::S) -> Self {
401        Node::<M>::leaf(val).into()
402    }
403
404    pub fn rank(&self) -> Option<usize> {
405        self.top.as_ref().map(|top| top.rank)
406    }
407
408    pub fn len(&self) -> usize {
409        self.top.as_ref().map_or(0, |top| top.len)
410    }
411
412    pub fn is_empty(&self) -> bool {
413        self.top.is_none()
414    }
415
416    pub fn merge(&self, rhs: &Self) -> Self {
417        match (&self.top, &rhs.top) {
418            (None, b) => b.clone().into(),
419            (a, None) => a.clone().into(),
420            (Some(a), Some(b)) => a.clone().merge(b.clone()).into(),
421        }
422    }
423
424    pub fn merge3(&self, m: &Self, r: &Self) -> Self {
425        if self.rank() < r.rank() {
426            self.merge(m).merge(r)
427        } else {
428            self.merge(&m.merge(r))
429        }
430    }
431
432    pub fn split(&self, p: usize) -> (Self, Self) {
433        assert!(p <= self.len());
434        if p == 0 {
435            return (Self::default(), self.clone());
436        }
437        if p == self.len() {
438            return (self.clone(), Self::default());
439        }
440        let (l, r) = self.top.clone().unwrap().split(p);
441        (l.into(), r.into())
442    }
443
444    pub fn split3(&self, range: Range<usize>) -> (Self, Self, Self) {
445        let (l, r) = (range.start, range.end);
446        assert!(r <= self.len());
447        if l == 0 {
448            let (m, r) = self.split(r);
449            return (Self::default(), m, r);
450        }
451        if r == self.len() {
452            let (l, m) = self.split(l);
453            return (l, m, Self::default());
454        }
455        if l == r {
456            let (l, r) = self.split(l);
457            return (l, Self::default(), r);
458        }
459        let (l, m, r) = self.top.clone().unwrap().split3(l, r);
460        (l.into(), m.into(), r.into())
461    }
462
463    pub fn insert(&self, p: usize, val: M::S) -> Self {
464        let (l, r) = self.split(p);
465        let m = Self::new(val);
466        l.merge3(&m, &r)
467    }
468
469    pub fn erase(&self, p: usize) -> Self {
470        let (l, _, r) = self.split3(p..p + 1);
471        l.merge(&r)
472    }
473
474    pub fn set(&self, p: usize, val: M::S) -> Self {
475        self.top.clone().map(|top| top.set(p, val)).into()
476    }
477
478    pub fn prod(&self, range: Range<usize>) -> M::S {
479        let (l, r) = (range.start, range.end);
480        self.top
481            .clone()
482            .map_or(M::e(), |top| top.prod(l, r, &M::id(), false))
483    }
484
485    pub fn apply(&self, range: Range<usize>, f: M::F) -> Self {
486        if f == M::id() {
487            return self.clone();
488        }
489        let (l, r) = (range.start, range.end);
490        self.top.clone().map(|top| top.apply(l, r, &f)).into()
491    }
492
493    pub fn reverse(&self, range: Range<usize>) -> Self {
494        let (l, m, r) = self.split3(range);
495        let m = m.top.map(|top| top.update(&M::id(), true)).into();
496        l.merge3(&m, &r)
497    }
498
499    pub fn dump(&self) -> Vec<M::S> {
500        let mut dump = Vec::with_capacity(self.len());
501        if let Some(top) = &self.top {
502            top.clone().dump(&mut dump);
503        }
504        dump
505    }
506}

508impl<M: MAct> FromIterator<M::S> for RedBlackTree<M> {
509    fn from_iter<T: IntoIterator<Item = M::S>>(iter: T) -> Self {
510        let data: Vec<_> = iter.into_iter().collect();
511        merge_rec(&data)
512    }
513}
514
515fn merge_rec<M: MAct>(data: &[M::S]) -> RedBlackTree<M> {
516    let n = data.len();
517    if n == 1 {
518        RedBlackTree::new(data[0].clone())
519    } else {
520        merge_rec(&data[..n / 2]).merge(&merge_rec(&data[n / 2..]))
521    }
522}
