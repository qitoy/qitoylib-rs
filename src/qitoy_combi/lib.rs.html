1use ac_library::modint::ModIntBase;
2
3/// 組み合わせで頻出の計算をModIntで計算します。
4pub struct Combi<T: ModIntBase> {
5    fact: Vec<T>,
6    factinv: Vec<T>,
7}
8
9impl<T: ModIntBase> Combi<T> {
10    /// `n`までの階乗とその逆元を前計算します。  
11    /// 計算量$`O(n)`$
12    pub fn new(n: usize) -> Self {
13        let modulus = T::modulus() as usize;
14        let mut fact = vec![T::new(1); n + 1];
15        let mut inv = vec![T::new(1); n + 1];
16        let mut factinv = vec![T::new(1); n + 1];
17        for i in 2..=n {
18            fact[i] = fact[i - 1] * i.into();
19            inv[i] = T::raw(modulus as u32) - inv[modulus % i] * (modulus / i).into();
20            factinv[i] = factinv[i - 1] * inv[i];
21        }
22        Self { fact, factinv }
23    }
24
25    /// 階乗を返します。  
26    /// 計算量$`O(1)`$
27    pub fn fact(&self, n: usize) -> T {
28        self.fact[n]
29    }
30
31    /// 階乗の逆元を返します。  
32    /// 計算量$`O(1)`$
33    pub fn factinv(&self, n: usize) -> T {
34        self.factinv[n]
35    }
36
37    /// $`{}_{n}P{}_{k}`$を返します。  
38    /// 計算量$`O(1)`$
39    pub fn perm(&self, n: usize, k: usize) -> T {
40        if n < k {
41            T::raw(0)
42        } else {
43            self.fact[n] * self.factinv[n - k]
44        }
45    }
46
47    /// $`{}_{n}C{}_{k}`$を返します。  
48    /// 計算量$`O(1)`$
49    pub fn combi(&self, n: usize, k: usize) -> T {
50        if n < k {
51            T::raw(0)
52        } else {
53            self.fact[n] * self.factinv[n - k] * self.factinv[k]
54        }
55    }
56}
