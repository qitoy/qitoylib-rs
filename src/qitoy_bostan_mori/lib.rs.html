3pub fn bostan_mori<M: Modulus>(
4    p: &[StaticModInt<M>],
5    q: &[StaticModInt<M>],
6    n: u64,
7) -> StaticModInt<M> {
8    if p.is_empty() {
9        return 0.into();
10    }
11
12    let even = |p: Vec<_>| p.into_iter().step_by(2).collect();
13    let odd = |p: Vec<_>| p.into_iter().skip(1).step_by(2).collect();
14
15    let (mut p, mut q, mut n) = (p.to_owned(), q.to_owned(), n);
16    while n > 0 {
17        let qm: Vec<StaticModInt<M>> = q
18            .iter()
19            .zip([0, 1].into_iter().cycle())
20            .map(|(&q, s)| if s == 0 { q } else { -q })
21            .collect();
22        let u = ac_library::convolution(&p, &qm);
23        p = if n & 1 == 0 { even(u) } else { odd(u) };
24        q = even(ac_library::convolution(&q, &qm));
25        n >>= 1;
26    }
27    p[0] / q[0]
28}
