3#[derive(Clone, Debug, Default, PartialEq, Eq)]
4pub struct Edge {
5    pub index: usize,
6    pub from: usize,
7    pub to: usize,
8}
9
10pub struct Tree {
11    /// (index, to)
12    data: Vec<Vec<Edge>>,
13    _cnt: usize,
14}
15
16impl Tree {
17    pub fn new(n: usize) -> Self {
18        Self {
19            data: vec![vec![]; n],
20            _cnt: 0,
21        }
22    }
23
24    pub fn add_egde(&mut self, v: usize, u: usize) {
25        self.data[v].push(Edge {
26            index: self._cnt,
27            from: v,
28            to: u,
29        });
30        self.data[u].push(Edge {
31            index: self._cnt,
32            from: u,
33            to: v,
34        });
35        self._cnt += 1;
36    }
37
38    pub fn is_valid(&self) -> bool {
39        self._cnt + 1 == self.data.len()
40    }
41}
42
43impl From<&Vec<(usize, usize)>> for Tree {
44    fn from(value: &Vec<(usize, usize)>) -> Self {
45        let mut tree = Self::new(value.len() + 1);
46        for &(v, u) in value {
47            tree.add_egde(v, u);
48        }
49        tree
50    }
51}
52
53impl FromIterator<(usize, usize)> for Tree {
54    fn from_iter<T: IntoIterator<Item = (usize, usize)>>(iter: T) -> Self {
55        let edges = iter.into_iter().collect();
56        Tree::from(&edges)
57    }
58}
