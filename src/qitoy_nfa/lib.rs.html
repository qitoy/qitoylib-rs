1use qitoy_dfa::Dfa;
2use std::collections::BTreeSet;
3
4/// 非決定性有限オートマトン
5pub trait Nfa {
6    /// 状態の型
7    type State: Ord + Clone;
8    /// 文字の型
9    type Alphabet;
10    /// 遷移関数
11    fn trans(&self, state: &Self::State, alpha: Option<&Self::Alphabet>) -> BTreeSet<Self::State>;
12    /// 開始状態
13    fn init(&self) -> BTreeSet<Self::State>;
14    /// 受理関数
15    fn accept(&self, state: &Self::State) -> bool;
16
17    fn epsilon_closure(&self, mut state: BTreeSet<Self::State>) -> BTreeSet<Self::State> {
18        let mut que = state.clone();
19        while let Some(x) = que.pop_first() {
20            state.insert(x.clone());
21            self.trans(&x, None)
22                .into_iter()
23                .filter(|y| !state.contains(y))
24                .for_each(|y| {
25                    que.insert(y);
26                });
27        }
28        state
29    }
30
31    #[inline]
32    fn to_dfa(self) -> ToDfa<Self>
33    where
34        Self: Sized,
35    {
36        ToDfa { nfa: self }
37    }
38}
39
40pub struct ToDfa<N> {
41    nfa: N,
42}
43
44impl<N: Nfa> Dfa for ToDfa<N> {
45    type State = BTreeSet<N::State>;
46    type Alphabet = N::Alphabet;
47
48    #[inline]
49    fn trans(&self, state: &Self::State, alpha: &Self::Alphabet) -> Option<Self::State> {
50        let ret = self.nfa.epsilon_closure(
51            state
52                .iter()
53                .map(|s| self.nfa.trans(s, Some(alpha)))
54                .fold(BTreeSet::new(), |a, x| &a | &x),
55        );
56        (!ret.is_empty()).then_some(ret)
57    }
58
59    #[inline]
60    fn init(&self) -> Self::State {
61        self.nfa.init()
62    }
63
64    #[inline]
65    fn accept(&self, state: &Self::State) -> bool {
66        state.iter().any(|s| self.nfa.accept(s))
67    }
68}
