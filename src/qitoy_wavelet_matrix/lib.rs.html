4/// Wavelet Matrix
5#[derive(Debug)]
6pub struct WaveletMatrix {
7    matrix: Vec<BitVec>,
8    zeros: Vec<usize>,
9}
10
11macro_rules! impl_from_iterator {
12    ($ty:ty, $e:expr) => {
13        impl FromIterator<$ty> for WaveletMatrix {
14            fn from_iter<T: IntoIterator<Item = $ty>>(iter: T) -> Self {
15                let data = iter.into_iter().map(u64::from).collect();
16                Self::new(data, $e)
17            }
18        }
19    };
20}
21
22impl_from_iterator!(u32, 32);
23impl_from_iterator!(u64, 64);
24
25impl WaveletMatrix {
26    /// max data < 1 << size
27    pub fn new(mut data: Vec<u64>, size: usize) -> Self {
28        let mut matrix = Vec::with_capacity(size);
29        let mut zeros = Vec::with_capacity(size);
30        for i in (0..size).rev() {
31            let bitvec: BitVec = data.iter().map(|v| v.bit_get(i)).collect();
32            zeros.push(bitvec.rank0(bitvec.len()));
33            matrix.push(bitvec);
34            data.sort_by_key(|v| v.bit_get(i));
35        }
36        Self { matrix, zeros }
37    }
38
39    fn size(&self) -> usize {
40        self.matrix.len()
41    }
42
43    pub fn len(&self) -> usize {
44        self.matrix[0].len()
45    }
46
47    pub fn is_empty(&self) -> bool {
48        self.len() == 0
49    }
50
51    pub fn access(&self, mut n: usize) -> u64 {
52        let mut ret = 0;
53        let size = self.size();
54        for i in 0..size {
55            let bitvec = &self.matrix[i];
56            if bitvec.at(n) {
57                n = self.zeros[i] + bitvec.rank1(n);
58                ret.bit_set(size - 1 - i);
59            } else {
60                n = bitvec.rank0(n);
61            }
62        }
63        ret
64    }
65
66    pub fn rank(&self, val: u64, n: usize) -> u32 {
67        (self.rank_pos(val, n) - self.rank_pos(val, 0)) as u32
68    }
69
70    fn rank_pos(&self, val: u64, mut n: usize) -> usize {
71        let size = self.size();
72        for i in 0..size {
73            let bitvec = &self.matrix[i];
74            n = if val.bit_get(size - 1 - i) {
75                self.zeros[i] + bitvec.rank1(n)
76            } else {
77                bitvec.rank0(n)
78            };
79        }
80        n
81    }
82}
